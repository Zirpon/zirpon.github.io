// {同花顺主力启动信号公式}

// {涨停识别}
涨停条件:= (CLOSE/REF(CLOSE,1)>=1.095);
二十日涨停:= COUNT(涨停条件,20)>=1;

// {涨停放量条件}
涨停放量:= 涨停条件 AND VOL>REF(VOL,1)*1.5;

// {排除缩量涨停}
非缩量涨停:= 涨停条件 AND VOL>=REF(VOL,1);

// {缺口识别}
跳空缺口:= LOW>REF(HIGH,1);
缺口形成日:= 跳空缺口 AND REF(跳空缺口,1)=0;
缺口天数:= BARSLAST(缺口形成日);
三天不回补:= 缺口天数<=3 AND LLV(LOW,缺口天数+1)>REF(HIGH,缺口天数+1);

// {连阳识别}
连阳天数:= BARSLAST(CLOSE<=OPEN);
优质连阳:= 连阳天数>=4 AND EVERY(CLOSE>REF(CLOSE,1),连阳天数) AND EVERY(LOW>REF(CLOSE,1),连阳天数) AND COUNT(VOL>REF(VOL,1),连阳天数)>=连阳天数/2;

// {关键位置倍量}
日均量:= MA(VOL,20);
倍量:= VOL>日均量*2;
平台高点:= HHV(HIGH,20);
突破平台:= CLOSE>REF(平台高点,1)*0.98;
前期高点:= HHV(HIGH,60);
突破前高:= CLOSE>REF(前期高点,1)*0.98;
关键位置:= 突破平台 OR 突破前高 OR COUNT(跳空缺口,5)>=1;
关键位置倍量:= 倍量 AND 关键位置;
站稳确认:= 关键位置倍量 AND COUNT(CLOSE>REF(CLOSE,1),3)>=2 AND HHV(HIGH,3)>REF(HIGH,1);

// {综合信号}
主力启动: (二十日涨停 AND 涨停放量 AND 三天不回补 AND 优质连阳 AND 站稳确认) OR (二十日涨停 AND 涨停放量 AND 三天不回补 AND 关键位置倍量 AND 站稳确认);

// {同花顺主力启动_分级信号}

// {基础条件}
涨停条件:= (CLOSE/REF(CLOSE,1)>=1.095);
二十日涨停:= COUNT(涨停条件,20)>=1;

// {缺口条件}
跳空缺口:= LOW>REF(HIGH,1);
缺口形成日:= 跳空缺口 AND REF(跳空缺口,1)=0;
缺口天数:= BARSLAST(缺口形成日);
三天不回补:= 缺口天数<=3 AND LLV(LOW,缺口天数+1)>REF(HIGH,缺口天数+1);

// {量能条件}
日均量:= MA(VOL,20);
倍量:= VOL>日均量*2;
涨停放量:= 涨停条件 AND VOL>REF(VOL,1)*1.5;

// {连阳条件}
连阳天数:= BARSLAST(CLOSE<=OPEN);
优质连阳:= 连阳天数>=3 AND EVERY(CLOSE>REF(CLOSE,1),连阳天数);

// {分级信号}
一级信号:= 二十日涨停 AND 三天不回补;
二级信号:= 一级信号 AND (涨停放量 OR 倍量);
三级信号:= 二级信号 AND 优质连阳;

// {输出信号}
信号级别: IF(三级信号,3,IF(二级信号,2,IF(一级信号,1,0)));

// {选股条件}
选股三级: 信号级别>=3;
选股二级: 信号级别>=2;
选股一级: 信号级别>=1;
