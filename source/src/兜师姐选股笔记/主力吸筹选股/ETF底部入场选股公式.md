// ETF底部入场选股公式 - 宽松版
// 核心理念：寻找处于相对低位的ETF，不追高，分批布局

// --- 1. 价格低位判断 (放宽核心条件) ---
// 计算12个月最低价（约250个交易日）
近250日最低价 := LLV(LOW, 250);
近250日最高价 := HHV(HIGH, 250);

// 放宽条件1：当前价格处于12个月最低价的1.15倍以内（即比最低价高不超过15%）
价格接近12月最低 := CLOSE <= 近250日最低价 * 1.15;

// 放宽条件2：当前价格处于12个月价格区间的低位（底部30%区域）
价格在底部区域 := CLOSE <= 近250日最低价 + (近250日最高价 - 近250日最低价) * 0.3;

// 条件3：跌幅条件放宽 - 从120日高点跌幅超过15%即可
跌幅足够大 := (HHV(HIGH, 120) - CLOSE) / HHV(HIGH, 120) > 0.15;

// 条件4：新增 - 距离最高点跌幅足够深（更直观）
距离最高点跌幅 := (HHV(HIGH, 250) - CLOSE) / HHV(HIGH, 250) * 100;
跌幅足够 := 距离最高点跌幅 > 20;  // 从250日高点下跌超过20%

// --- 2. 超卖指标确认 (放宽) ---
RSV := (CLOSE - LLV(LOW, 9)) / (HHV(HIGH, 9) - LLV(LOW, 9)) * 100;
K值 := SMA(RSV, 3, 1);
D值 := SMA(K值, 3, 1);
J值 := 3*K值 - 2*D值;

// 放宽超卖条件：J值小于40就算超卖区域
J值超卖 := J值 < 40;
J值底部区域 := J值 < 50 AND J值 > REF(J值, 1); // J值在50以下且开始拐头向上
J值底部金叉 := CROSS(J值, D值) AND J值 < 50;

// --- 3. 成交量确认 (放宽) ---
VOL5 := MA(VOL, 5);
VOL10 := MA(VOL, 10);

// 放宽放量条件：成交量不是极度萎缩即可
底部放量 := VOL > VOL5 * 1.1;  // 只要大于5日均量10%即可
量能不萎缩 := VOL5 > REF(VOL5, 10) * 0.8;  // 5日均量不低于10天前的80%

// --- 4. 趋势条件 (放宽/取消) ---
MA20 := MA(CLOSE, 20);
MA60 := MA(CLOSE, 60);

// 改为：不要求远离套牢区，只要不过分高于均线即可
不追高 := CLOSE <= MA20 * 1.05;  // 价格不高于20日线5%
// 或者直接取消这个条件

// --- 5. 新增：底部横盘或温和反弹 ---
// 最近20天波动幅度小（横盘或筑底）
近期振幅 := (HHV(HIGH, 20) - LLV(LOW, 20)) / LLV(LOW, 20) * 100;
底部横盘 := 近期振幅 < 25 AND CLOSE > LLV(LOW, 20) * 1.05;  // 振幅小于25%，且脱离最低点

// 最近5天有企稳迹象
企稳天数 := COUNT(CLOSE > OPEN, 3) >= 2;  // 3天中2天收阳线
或者 := COUNT(CLOSE > REF(CLOSE, 1), 4) >= 2;  // 4天中2天上涨

// --- 综合选股条件 (多个宽松组合) ---

// 核心条件：价格在相对低位
核心底部条件 := 价格接近12月最低 OR 价格在底部区域;

// 跌幅确认（二选一）
跌幅确认 := 跌幅足够大 OR 跌幅足够;

// --- 提供多个宽松选股方案 ---

// 方案1：超宽松版 - 只要价格够低就行，其他条件不强制
ETF超宽松 := 核心底部条件 AND J值 < 60;  // 只要J值不在绝对高位

// 方案2：温和版 - 价格低 + 超卖 + 跌幅够
ETF温和版 := 核心底部条件 AND J值超卖 AND 跌幅确认;

// 方案3：企稳版 - 价格低 + 企稳迹象 + 放量
ETF企稳版 := 核心底部条件 AND 企稳天数 AND 底部放量 AND J值 < 60;

// 方案4：推荐使用 - 综合平衡版（最容易选出股票）
ETF综合版 := (核心底部条件) AND 
            (J值 < 55) AND  // J值在55以下
            (跌幅确认 OR 距离最高点跌幅 > 15) AND  // 跌幅超过15%
