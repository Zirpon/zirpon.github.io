

RSV:=(CLOSE-LLV(LOW,9))/(HHV(HIGH,9)-LLV(LOW,9))*100;
KDJ_K:SMA(RSV,3,1),COLORMAGENTA;
KDJ_D:SMA(KDJ_K,3,1),COLORBLUE;
KDJ_J:3*KDJ_K-2*KDJ_D,COLORLIBLUE;

MACD_DIFF:=EMA(CLOSE,12)-EMA(CLOSE,26);
MACD_DEA:=EMA(MACD_DIFF,9);

水下金叉1:=CROSS(MACD_DIFF,MACD_DEA) AND MACD_DIFF<0 AND MACD_DEA<0 AND CROSS(KDJ_J,KDJ_D) AND KDJ_J<30;
水下死叉1:=CROSS(MACD_DEA,MACD_DIFF) AND MACD_DIFF<0 AND MACD_DEA<0 AND CROSS(KDJ_D,KDJ_J) AND KDJ_D<30;
水上金叉1:=CROSS(MACD_DIFF,MACD_DEA) AND MACD_DIFF>0 AND MACD_DEA>0 AND CROSS(KDJ_J,KDJ_D) AND KDJ_J>70;
水上死叉1:=CROSS(MACD_DEA,MACD_DIFF) AND MACD_DIFF>0 AND MACD_DEA>0 AND CROSS(KDJ_D,KDJ_J) AND KDJ_D>70;

// 提前预警版本（金叉前预兆）
KDJ金叉预兆:=CROSS(KDJ_J,KDJ_D) AND KDJ_J<30;
MACD金叉预兆:=MACD_DIFF>MACD_DEA AND REF(MACD_DIFF,1)<REF(MACD_DEA,1) AND MACD_DIFF<0;
水下金叉预警:=KDJ金叉预兆 AND MACD金叉预兆;


// 显示不同级别的信号
DRAWICON(水下金叉1,KDJ_J*0.95,1);
DRAWTEXT(水下金叉1,KDJ_J*1.2,'上涨'),COLORRED;

DRAWICON(水下死叉1,KDJ_J*0.95,2);
DRAWTEXT(水下死叉1,KDJ_J*0.8,'看空'),COLORGREEN;

DRAWICON(水上金叉1,KDJ_J*1.05,11);
DRAWTEXT(水上金叉1,KDJ_J*1.2,'大涨'),COLORMAGENTA;

DRAWICON(水上死叉1,KDJ_J*0.95,12);
DRAWTEXT(水上死叉1,KDJ_J*0.8,'强空'),COLORBLUE;

// 预警信号用不同图标
DRAWICON(水下金叉预警,KDJ_J*0.9,5);
DRAWTEXT(水下金叉预警,KDJ_J*1.1,'金叉预警'),COLORYELLOW;
